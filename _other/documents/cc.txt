-- DOCKER
-- Lifecycle & State Commands
		Start everything in background (most common way to run in production/dev):text
		docker compose up -d

		Stop everything gracefully (without removing containers/volumes/networks):text
		docker compose stop

		Start already-created services (after a previous down or stop):text
		docker compose start

		Restart one or all services (without full recreation):text
		docker compose restart
		docker compose restart app      # only the 'app' service

		Full clean restart (your variant, but shorter/more common):text
		docker compose down && docker compose up -d

		or with volumes wipe
		docker compose down -v && docker compose up -d

-- Status & Inspection
		List running services/containers (very frequently used):textdocker compose ps
		docker compose ps -a          # also show stopped/exited ones

		Show real-time resource usage:textdocker compose top
		docker compose stats

-- Logs & Debugging

		View logs of all services (or specific one)
		docker compose logs
		docker compose logs -f                # follow/tail
		docker compose logs -f --tail=100 app # last 100 lines + follow

		Timestamps + colors (easier to read)
		docker compose logs -f --timestamps

-- Exec & Running Commands
		Run a one-off command (without starting the service permanently)
		docker compose run --rm app python manage.py migrate
		docker compose run --rm app bash      # get a shell
		
		Exec into a running container (like your psql example)
		docker compose exec app bash          # interactive shell
		docker compose exec app sh -c "echo hello"

-- Build & Pull
		Rebuild images (when Dockerfile or dependencies changed):text
		docker compose build
		docker compose build --no-cache app   # force full rebuild

		Pull latest images (without rebuilding):text
		docker compose pull

-- Cleanup & Maintenance
		Remove stopped containers, networks & dangling images:text
		docker compose down --remove-orphans

		Prune everything not in use (including volumes â€” be careful!):text
		docker compose down --volumes --remove-orphans
		docker system prune -a --volumes      # global cleanup (not compose-specific)
